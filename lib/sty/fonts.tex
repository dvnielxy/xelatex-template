% FONT CONFIGURATION
%%%%%%%%%%%%%%%%%%%%

% Montserrat font
\newfontfamily\montserratfamily{Montserrat}
\newenvironment{montserrat}{\montserratfamily}{\par}
\DeclareTextFontCommand{\textmontserrat}{\montserratfamily}

% Corbel font
\newfontfamily\corbelfamily{Corbel}[Numbers=Lining]
\newenvironment{corbel}{\corbelfamily}{\par}
\DeclareTextFontCommand{\textcorbel}{\corbelfamily}

% FiraCode font
\newfontfamily\firacodefamily{Fira Code}
\newenvironment{firacode}{\firacodefamily}{\par}
\DeclareTextFontCommand{\textfiracode}{\firacodefamily}

% Latin Modern Roman font
\newfontfamily\LMRfamily{Latin Modern Roman}
\newenvironment{LMR}{\LMRfamily}{\par}
\DeclareTextFontCommand{\textLMR}{\LMRfamily}

% Latin Modern Sans font
\newfontfamily\LMSfamily{Latin Modern Sans}
\newenvironment{LMS}{\LMSfamily}{\par}
\DeclareTextFontCommand{\textLMS}{\LMSfamily}
